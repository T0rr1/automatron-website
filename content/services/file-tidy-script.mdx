## Sample Implementation

```powershell
# File Tidy Script - Safe Organization
param(
    [string]$SourcePath = "C:\Users\$env:USERNAME\Desktop",
    [switch]$DryRun = $true,
    [switch]$CreateBackup = $false
)

# Organization rules (customizable)
$OrganizationRules = @{
    "Documents" = @{
        Extensions = @(".pdf", ".docx", ".xlsx", ".pptx")
        Destination = "C:\Users\$env:USERNAME\Documents\Organized"
        SubfolderBy = "FileType"
    }
    "Images" = @{
        Extensions = @(".jpg", ".png", ".gif", ".bmp")
        Destination = "C:\Users\$env:USERNAME\Pictures\Organized"
        SubfolderBy = "Date"
    }
    "Archives" = @{
        Extensions = @(".zip", ".rar", ".7z")
        Destination = "C:\Users\$env:USERNAME\Downloads\Archives"
        SubfolderBy = "None"
    }
}

# Safety check and backup
if ($CreateBackup) {
    $backupPath = "C:\FileBackups\$(Get-Date -Format 'yyyy-MM-dd_HH-mm')"
    Write-Host "Creating backup at: $backupPath"
    Copy-Item $SourcePath -Destination $backupPath -Recurse
}

# Process files safely
foreach ($rule in $OrganizationRules.GetEnumerator()) {
    $category = $rule.Key
    $config = $rule.Value
    
    Get-ChildItem -Path $SourcePath -File | 
    Where-Object { $config.Extensions -contains $_.Extension.ToLower() } |
    ForEach-Object {
        $destFolder = Get-DestinationFolder -File $_ -Config $config
        
        if ($DryRun) {
            Write-Host "[DRY RUN] Would move: $($_.Name) -> $destFolder" -ForegroundColor Yellow
        } else {
            if (-not (Test-Path $destFolder)) {
                New-Item -ItemType Directory -Path $destFolder -Force | Out-Null
            }
            Move-Item $_.FullName -Destination $destFolder
            Write-Host "Moved: $($_.Name) -> $destFolder" -ForegroundColor Green
        }
    }
}
```

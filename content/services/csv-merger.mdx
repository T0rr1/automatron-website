## Sample Implementation

```python
import pandas as pd
import glob
import os
from datetime import datetime

def merge_csv_files(input_folder, output_file, config):
    """
    Merge multiple CSV files with cleaning and validation
    """
    print(f"Processing CSV files from: {input_folder}")
    
    # Find all CSV files
    csv_files = glob.glob(os.path.join(input_folder, "*.csv"))
    
    if not csv_files:
        raise ValueError("No CSV files found in the specified folder")
    
    merged_data = []
    
    for file in csv_files:
        print(f"Processing: {os.path.basename(file)}")
        
        try:
            # Read CSV with error handling
            df = pd.read_csv(file, encoding='utf-8')
            
            # Apply column mapping if specified
            if 'column_mapping' in config:
                df = df.rename(columns=config['column_mapping'])
            
            # Add source file information
            df['source_file'] = os.path.basename(file)
            df['processed_date'] = datetime.now().strftime('%Y-%m-%d %H:%M')
            
            # Data validation and cleaning
            df = clean_data(df, config)
            
            merged_data.append(df)
            
        except Exception as e:
            print(f"Error processing {file}: {str(e)}")
            continue
    
    if not merged_data:
        raise ValueError("No valid data found in CSV files")
    
    # Combine all dataframes
    final_df = pd.concat(merged_data, ignore_index=True)
    
    # Remove duplicates if specified
    if config.get('remove_duplicates', True):
        duplicate_cols = config.get('duplicate_columns', None)
        final_df = final_df.drop_duplicates(subset=duplicate_cols)
    
    # Sort data if specified
    if 'sort_columns' in config:
        final_df = final_df.sort_values(config['sort_columns'])
    
    # Save merged file
    final_df.to_csv(output_file, index=False)
    
    print(f"Merged {len(csv_files)} files into {output_file}")
    print(f"Total rows: {len(final_df)}")
    
    return final_df

def clean_data(df, config):
    """Clean and validate data"""
    
    # Remove empty rows
    df = df.dropna(how='all')
    
    # Standardize date columns
    date_columns = config.get('date_columns', [])
    for col in date_columns:
        if col in df.columns:
            df[col] = pd.to_datetime(df[col], errors='coerce')
    
    # Standardize numeric columns
    numeric_columns = config.get('numeric_columns', [])
    for col in numeric_columns:
        if col in df.columns:
            df[col] = pd.to_numeric(df[col], errors='coerce')
    
    # Clean text columns
    text_columns = config.get('text_columns', [])
    for col in text_columns:
        if col in df.columns:
            df[col] = df[col].astype(str).str.strip().str.title()
    
    return df

# Example configuration
config = {
    'column_mapping': {
        'Customer Name': 'customer_name',
        'Sale Amount': 'amount',
        'Sale Date': 'date'
    },
    'date_columns': ['date'],
    'numeric_columns': ['amount'],
    'text_columns': ['customer_name'],
    'remove_duplicates': True,
    'duplicate_columns': ['customer_name', 'date'],
    'sort_columns': ['date', 'amount']
}

# Run the merger
if __name__ == "__main__":
    merge_csv_files("./input_csvs", "./merged_output.csv", config)
```
